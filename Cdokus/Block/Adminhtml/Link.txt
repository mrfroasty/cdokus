namespace Zanbytes\Cdokus\Block\Adminhtml;

/**
 * Adminhtml cms pages content block
 */
class Link extends \Magento\Backend\Block\Widget\Grid\Container
{
    /**
     * Block constructor
     *
     * @return void
     */
    protected function _construct()
    {
        $this->_controller = 'adminhtml_link';
        $this->_blockGroup = 'Zanbytes_Cdokus';
        $this->_headerText = __('Manage Pages');

        parent::_construct();

        if ($this->_isAllowedAction('Zanbytes_Cdokus::save')) {
            $this->buttonList->update('add', 'label', __('Add New Document'));
        } else {
            $this->buttonList->remove('add');
        }
    }

    /**
     * Check permission for passed action
     *
     * @param string $resourceId
     * @return bool
     */
    protected function _isAllowedAction($resourceId)
    {
        return $this->_authorization->isAllowed($resourceId);
    }
}
